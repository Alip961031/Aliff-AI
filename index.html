// ============================================================ // ALIFF PROXY – FINAL WORKER // Features: KV history, R2 uploads, streaming, search, developer kit // ============================================================ const DEEPSEEK_URL = "https://api.deepseek.com/v1/chat/completions"; const SEARX_INSTANCES = ["https://searx.be", "https://search.bus-hit.me"]; function corsHeaders(request) { const origin = request.headers.get("Origin") || "*"; return { "Access-Control-Allow-Origin": origin, "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type, X-Session-Id", "Vary": "Origin" }; } // ---------- KV history ---------- async function getHistory(env, sessionId) { const key = `history:${sessionId}`; const data = await env.ALIFF_MEMORY.get(key); return data ? JSON.parse(data) : []; } async function saveHistory(env, sessionId, messages) { const key = `history:${sessionId}`; await env.ALIFF_MEMORY.put(key, JSON.stringify(messages), { expirationTtl: 86400 * 7 }); } // ---------- DeepSeek API ---------- async function callDeepSeek(env, messages, stream = false) { const response = await fetch(DEEPSEEK_URL, { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${env.DEEPSEEK_API_KEY}` }, body: JSON.stringify({ model: "deepseek-chat", messages, stream, temperature: 0.7, max_tokens: 2000, safe_mode: env.SAFE_MODE === "true" }) }); if (!response.ok) { const errorText = await response.text(); throw new Error(`DeepSeek error ${response.status}: ${errorText}`); } return response; } // ---------- Internet search (SearXNG) ---------- async function searchInternet(query) { const params = new URLSearchParams({ q: query, format: "json" }); for (const instance of SEARX_INSTANCES) { try { const url = `${instance}/search?${params}`; const res = await fetch(url, { headers: { "User-Agent": "Aliff/1.0" } }); if (!res.ok) continue; const data = await res.json(); const results = data.results || []; if (results.length) { return results.slice(0, 2).map(r => `- ${r.title}\n ${r.url}\n ${r.content?.slice(0, 300)}`).join("\n\n"); } } catch (e) {} } return null; } export default { async fetch(request, env, ctx) { const url = new URL(request.url); const path = url.pathname; // CORS preflight if (request.method === "OPTIONS") { return new Response(null, { status: 204, headers: corsHeaders(request) }); } // Health check if (request.method === "GET" && path === "/health") { return new Response(JSON.stringify({ status: "ok" }), { headers: corsHeaders(request) }); } // 1. File upload to R2 if (request.method === "POST" && path === "/upload") { try { const formData = await request.formData(); const file = formData.get("file"); const sessionId = formData.get("sessionId"); if (!file || !sessionId) throw new Error("Missing file or sessionId"); const key = `${sessionId}/${Date.now()}-${file.name}`; await env.ATTACHMENTS.put(key, file.stream(), { httpMetadata: { contentType: file.type } }); return new Response(JSON.stringify({ ok: true, key }), { headers: corsHeaders(request) }); } catch (err) { return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: corsHeaders(request) }); } } // 2. Get conversation history if (request.method === "GET" && path === "/history") { const sessionId = url.searchParams.get("sessionId"); if (!sessionId) return new Response(JSON.stringify({ error: "Missing sessionId" }), { status: 400, headers: corsHeaders(request) }); const history = await getHistory(env, sessionId); return new Response(JSON.stringify({ history }), { headers: corsHeaders(request) }); } // 3. Chat endpoint if (request.method === "POST" && path === "/chat") { try { const body = await request.json(); let { message, sessionId, attachments = [], stream = true, systemPrompt, useInternetSearch } = body; if (!message && attachments.length === 0) { return new Response(JSON.stringify({ error: "No message or attachments" }), { status: 400, headers: corsHeaders(request) }); } // Load existing conversation let conversation = await getHistory(env, sessionId); // Internet search let searchContext = ""; if (useInternetSearch && message) { const results = await searchInternet(message); if (results) searchContext = `\n\n[Internet search results]\n${results}\n`; } let userContent = message; if (attachments.length) userContent += `\n\n[Attachments: ${attachments.join(", ")}]`; if (searchContext) userContent += searchContext; const finalSystemPrompt = systemPrompt || "You are Aliff, a helpful, factual AI assistant. Keep answers short and clear."; const apiMessages = [ { role: "system", content: finalSystemPrompt }, ...conversation, { role: "user", content: userContent } ]; if (stream) { const deepseekRes = await callDeepSeek(env, apiMessages, true); const { readable, writable } = new TransformStream(); const writer = writable.getWriter(); const encoder = new TextEncoder(); let fullReply = ""; (async () => { const reader = deepseekRes.body.getReader(); const decoder = new TextDecoder(); try { while (true) { const { done, value } = await reader.read(); if (done) break; const chunk = decoder.decode(value); const lines = chunk.split("\n"); for (const line of lines) { if (line.startsWith("data: ")) { const data = line.slice(6); if (data === "[DONE]") { await writer.write(encoder.encode(`data: {"token":"","done":true}\n\n`)); break; } try { const parsed = JSON.parse(data); const token = parsed.choices?.[0]?.delta?.content || ""; if (token) { fullReply += token; await writer.write(encoder.encode(`data: ${JSON.stringify({ token })}\n\n`)); } } catch (e) {} } } } const newConversation = [ ...conversation, { role: "user", content: userContent }, { role: "assistant", content: fullReply } ]; await saveHistory(env, sessionId, newConversation); } catch (err) { console.error("Streaming error", err); } finally { await writer.close(); } })(); return new Response(readable, { headers: { "Content-Type": "text/event-stream", "Cache-Control": "no-cache", ...corsHeaders(request) } }); } else { const deepseekRes = await callDeepSeek(env, apiMessages, false); const data = await deepseekRes.json(); const reply = data.choices?.[0]?.message?.content || "Sorry, I couldn't generate a response."; const newConversation = [ ...conversation, { role: "user", content: userContent }, { role: "assistant", content: reply } ]; await saveHistory(env, sessionId, newConversation); return new Response(JSON.stringify({ reply }), { headers: corsHeaders(request) }); } } catch (err) { console.error(err); return new Response(JSON.stringify({ error: err.message }), { status: 500, headers: corsHeaders(request) }); } } return new Response("Not found", { status: 404, headers: corsHeaders(request) }); } };
